CREATE TABLE dynamic_pages (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,

    folder_name VARCHAR(100) NOT NULL,
    route_link VARCHAR(150) NULL,
    title VARCHAR(255) NULL,
    url VARCHAR(255) NULL,
    info TEXT NULL,

    is_active TINYINT(1) DEFAULT 1,
    is_enabled TINYINT(1) DEFAULT 1,
    status ENUM('active','inactive','blocked') NOT NULL DEFAULT 'active',

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE dynamic_tables (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,

    dynamic_page_id BIGINT UNSIGNED NOT NULL,
    table_name VARCHAR(150) NOT NULL UNIQUE,

    is_active TINYINT(1) DEFAULT 1,
    status ENUM('active','inactive','blocked') DEFAULT 'active',

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    CONSTRAINT fk_dynamic_tables_page
        FOREIGN KEY (dynamic_page_id)
        REFERENCES dynamic_pages(id)
        ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE dynamic_fields (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,

    dynamic_table_id BIGINT UNSIGNED NOT NULL,   -- FIXED (FIELD BELONGS TO TABLE)

    label VARCHAR(150) NULL,
    field_name VARCHAR(150) NOT NULL,
    field_type VARCHAR(100) NOT NULL,

    placeholder VARCHAR(255) NULL,
    is_required TINYINT(1) DEFAULT 0,

    is_active TINYINT(1) DEFAULT 1,
    is_enabled TINYINT(1) DEFAULT 1,
    status ENUM('active','inactive','blocked') DEFAULT 'active',

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    CONSTRAINT fk_dynamic_fields_table
        FOREIGN KEY (dynamic_table_id)
        REFERENCES dynamic_tables(id)
        ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE dynamic_field_options (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,

    field_id BIGINT UNSIGNED NOT NULL,

    option_value VARCHAR(100) NOT NULL,
    option_label VARCHAR(100) NOT NULL,
    is_default TINYINT(1) DEFAULT 0,

    is_active TINYINT(1) DEFAULT 1,
    is_enabled TINYINT(1) DEFAULT 1,
    status ENUM('active','inactive','blocked') DEFAULT 'active',

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    CONSTRAINT fk_field_options_field
        FOREIGN KEY (field_id)
        REFERENCES dynamic_fields(id)
        ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE dynamic_table_records (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    dynamic_table_id BIGINT UNSIGNED NOT NULL,
    submitted_by BIGINT UNSIGNED NULL,
    data JSON NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (dynamic_table_id) REFERENCES dynamic_tables(id) ON DELETE CASCADE
);
